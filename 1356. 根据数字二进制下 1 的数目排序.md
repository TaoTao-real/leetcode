'''C++
class Solution {
public:
    int static onecount(int num){
        int count = 0;
        while(num > 0){
            if((num & 1) == 1) count++;
            num >>= 1;
        }
        return count;
    }
    bool static cmp(int& a, int& b){
        int counta = onecount(a);
        int countb = onecount(b);
        return counta == countb?a < b:counta<countb;
    }
    vector<int> sortByBits(vector<int>& arr) {
        sort(arr.begin(), arr.end(), cmp);
        return arr;
    }
};
